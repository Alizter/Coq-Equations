Equations ack (m n : nat) : nat by rec (m, n) (lexprod _ _ lt lt) :=
  ack 0 0 := 1;
  ack 0 (S n) := S (S n);
  ack (S m) 0 := ack m 1;
  ack (S m) (S n) := ack m (ack (S m) n).

Fixpoint measure_abc (x : abc) :=
  match x with
  | abc0 ⇒ 0
  | A x ⇒ 3 + measure_abc x
  | B x ⇒ S (measure_abc x)
  | C x ⇒ S (measure_abc x)
  end.

Equations sct0 (x : abc) : nat by rec (measure_abc x) lt :=
  sct0 abc0 := 0;
  sct0 (A x) := sct0 (B (C x)) + sct0 x;
  sct0 (B x) := sct0 x;
  sct0 (C x) := sct0 x.

  | abc0 ⇒ 0
